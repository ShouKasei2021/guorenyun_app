# guorenyun_app

└── styles/
    ├── base/
    │   ├── _reset.scss     # CSS 重置
    │   ├── _variables.scss # 变量定义
    │   └── _mixins.scss    # 混入
    ├── components/         # 组件样式

# 运行项目
npm run dev

npm run build

├── tsconfig.json       # TypeScript 配置
└── vue.config.js       # Vue 配置
